Folder structure:
 * docker-compose.yml
 * create_table.sql
 * data (empty folder)

========================
FILE: docker-compose.yml
========================

version: '3.1'

services:
  rider-db:
    image: postgres:15
    container_name: rider_db
    restart: always
    ports:
      - 5432:5432
    volumes:
      - ./data:/var/lib/postgresql/data
      - ./create_tables.sql:/docker-entrypoint-initdb.d/create_tables.sql
    environment:
      POSTGRES_DB: root
      POSTGRES_USER: root
      POSTGRES_PASSWORD: postgres

=======================
FILE: create_tables.sql
=======================

create table if not exists riders (
id bigserial primary key,
name varchar(50) not null,
phone varchar(16) unique not null,
email varchar(50) unique,
status varchar(12) default 'unavailable',
password varchar(100) not null);

create table if not exists orders (
id bigserial primary key,
items varchar(200),
restaurant_details varchar(100),
restaurant_location_x numeric,
restaurant_location_y numeric,
customer_details varchar(100),
customer_location_x numeric,
customer_location_y numeric,
status varchar(12) default 'placed',
placed_at timestamp default current_timestamp,
delivered_at timestamp,
rider_id bigint references riders(id),
bill_amt numeric,
payment_mode varchar(20));
